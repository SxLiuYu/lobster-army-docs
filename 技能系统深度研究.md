## 🎯 技能系统概述

### 什么是技能（Skill）

```
Skill = 能力单元 = Agent 的工具包

一个 Skill 包含：
- name: 技能名称
- description: 描述
- triggers: 触发条件
- entry_point: 执行入口
- capabilities: 能力列表
```

### 技能 vs 工具

| 维度 | 工具 (Tool) | 技能 (Skill) |
|------|-------------|---------------|
| 定义 | 基础执行单元 | 能力封装 |
| 粒度 | 细 | 粗 |
| 触发 | 显式调用 | 自动 + 显式 |
| 组合 | 单一 | 多工具组合 |

## ⚙️ 技能加载机制

### 1. 自动发现

```python
class SkillLoader:
    """技能加载器"""
    
    def __init__(self, skills_dir: str = "./skills"):
        self.skills_dir = skills_dir
        self.registry = {}
    
    def auto_discover(self):
        """自动发现技能"""
        
        import os
        import yaml
        
        for root, dirs, files in os.walk(self.skills_dir):
            for file in files:
                if file.endswith((".yaml", ".yml", ".json")):
                    path = os.path.join(root, file)
                    self._load_skill(path)
    
    def _load_skill(self, path: str):
        """加载单个技能"""
        
        if path.endswith((".yaml", ".yml")):
            with open(path) as f:
                config = yaml.safe_load(f)
        elif path.endswith(".json"):
            with open(path) as f:
                config = json.load(f)
        
        skill = Skill(
            name=config["name"],
            description=config["description"],
            version=config.get("version", "1.0.0"),
            triggers=config.get("triggers", {}),
            execution=config.get("execution", {}),
            capabilities=config.get("capabilities", [])
        )
        
        self.registry[skill.name] = skill
        print(f"✅ 加载技能: {skill.name}")
    
    def get_skill(self, name: str) -> Skill:
        return self.registry.get(name)
    
    def list_skills(self) -> list[Skill]:
        return list(self.registry.values())
```

### 2. 动态注册

```python
class SkillRegistry:
    """技能注册表"""
    
    def __init__(self):
        self.skills = {}
        self.listeners = []
    
    def register(self, skill: Skill, auto_load: bool = True):
        """注册技能"""
        
        self.skills[skill.name] = skill
        
        # 注册触发器
        for keyword in skill.triggers.get("keywords", []):
            self._register_keyword(keyword, skill.name)
        
        # 通知 listeners
        for listener in self.listeners:
            listener.on_skill_registered(skill)
        
        print(f"✅ 注册技能: {skill.name}")
    
    def _register_keyword(self, keyword: str, skill_name: str):
        """注册关键词"""
        
        if not hasattr(self, "_keyword_map"):
            self._keyword_map = {}
        
        if keyword not in self._keyword_map:
            self._keyword_map[keyword] = []
        
        self._keyword_map[keyword].append(skill_name)
    
    def find_by_keyword(self, keyword: str) -> list[Skill]:
        """根据关键词查找技能"""
        
        skill_names = self._keyword_map.get(keyword, [])
        return [self.skills[name] for name in skill_names]
    
    def add_listener(self, listener):
        """添加监听器"""
        self.listeners.append(listener)
```

## 🔔 技能触发机制

### 1. 关键词触发

```python
class KeywordTrigger:
    """关键词触发器"""
    
    def __init__(self, registry: SkillRegistry):
        self.registry = registry
    
    def match(self, message: str) -> list[Skill]:
        """匹配技能"""
        
        message_lower = message.lower()
        matched = []
        
        for skill in self.registry.list_skills():
            keywords = skill.triggers.get("keywords", [])
            
            for keyword in keywords:
                if keyword.lower() in message_lower:
                    matched.append(skill)
                    break
        
        return matched
```

### 3. 事件触发

```python
class EventTrigger:
    """事件触发器"""
    
    def __init__(self):
        self.handlers = defaultdict(list)
    
    def on(self, event: str, handler: callable):
        """监听事件"""
        self.handlers[event].append(handler)
    
    async def emit(self, event: str, data: dict):
        """触发事件"""
        
        for handler in self.handlers[event]:
            await handler(data)


# 使用
trigger = EventTrigger()

# 定时事件
trigger.on("daily_morning", send_briefing)
trigger.on("hourly_check", health_check)

# 用户事件
trigger.on("user_login", welcome_user)
trigger.on("task_completed", notify_user)
```

### 4. 时间触发

```python
class ScheduleTrigger:
    """定时触发器"""
    
    def __init__(self):
        self.jobs = []
    
    def schedule(self, cron: str, task: callable):
        """添加定时任务"""
        
        self.jobs.append({
            "cron": cron,
            "task": task,
            "next_run": self._parse_cron(cron)
        })
    
    async def start(self):
        """启动定时器"""
        
        while True:
            now = datetime.now()
            
            for job in self.jobs:
                if now >= job["next_run"]:
                    await job["task"]()
                    job["next_run"] = self._parse_cron(job["cron"])
            
            await asyncio.sleep(60)


# 使用
scheduler = ScheduleTrigger()

# 每天早上9点
scheduler.schedule("0 9 * * *", daily_briefing)

# 每小时
scheduler.schedule("0 * * * *", hourly_health_check)

# 启动
await scheduler.start()

class SkillMarket:
    """技能市场"""
    
    def __init__(self):
        self.skills = {}
        self.categories = defaultdict(list)
    
    def publish(self, skill: Skill, category: str):
        """发布技能"""
        
        self.skills[skill.name] = skill
        self.categories[category].append(skill)
    
    def search(self, query: str, category: str = None) -> list[Skill]:
        """搜索技能"""
        
        results = []
        
        if category:
            candidates = self.categories.get(category, [])
        else:
            candidates = self.skills.values()
        
        for skill in candidates:
            if query.lower() in skill.name.lower() or \
               query.lower() in skill.description.lower():
                results.append(skill)
        
        return results
    
    def install(self, skill_name: str):
        """安装技能"""
        
        skill = self.skills.get(skill_name)
        if skill:
            SkillRegistry().register(skill)
            print(f"✅ 安装技能: {skill_name}")
        else:
            print(f"❌ 技能不存在: {skill_name}")
```

## 💻 实现示例

### 完整技能系统

```python
# skill_system.py
import yaml
import json
import asyncio
from dataclasses import dataclass
from typing import Optional, Callable

@dataclass
class Skill:
    """技能定义"""
    name: str
    description: str
    version: str = "1.0.0"
    triggers: dict = None
    execution: dict = None
    capabilities: list = None
    entry_point: Optional[Callable] = None
    
    def __post_init__(self):
        self.triggers = self.triggers or {}
        self.execution = self.execution or {}
        self.capabilities = self.capabilities or []


class SkillSystem:
    """技能系统"""
    
    def __init__(self):
        self.registry = {}
        self.loader = SkillLoader()
        self.trigger_handlers = [
            KeywordTrigger(self),
            # IntentTrigger(llm),
            # EventTrigger(),
        ]
    
    def load_skills_from_dir(self, path: str):
        """从目录加载技能"""
        self.loader.auto_discover()
        self.registry = self.loader.registry
    
    async def match_skills(self, message: str, context: dict = None) -> list[Skill]:
        """匹配技能"""
        
        matched = []
        
        for handler in self.trigger_handlers:
            results = handler.match(message, context or {})
            matched.extend(results)
        
        # 去重
        seen = set()
        unique = []
        for skill in matched:
            if skill.name not in seen:
                seen.add(skill.name)
                unique.append(skill)
        
        return unique
    
    async def execute_skill(self, skill: Skill, params: dict) -> dict:
        """执行技能"""
        
        timeout = skill.execution.get("timeout", 30)
        retry = skill.execution.get("retry", 0)
        
        for attempt in range(retry + 1):
            try:
                if skill.entry_point:
                    result = await asyncio.wait_for(
                        skill.entry_point(params),
                        timeout=timeout
                    )
                    return {"success": True, "result": result}
                    
            except asyncio.TimeoutError:
                if attempt == retry:
                    return {"success": False, "error": "timeout"}
                    
            except Exception as e:
                if attempt == retry:
                    return {"success": False, "error": str(e)}
        
        return {"success": False, "error": "max retries"}
    
    def register_skill(self, skill: Skill):
        """注册技能"""
        self.registry[skill.name] = skill
```

### 使用示例

```python
# main.py
async def main():
    # 初始化技能系统
    system = SkillSystem()
    
    # 加载技能
    system.load_skills_from_dir("./skills")
    
    # 处理消息
    message = "帮我搜一下最新的 AI 新闻"
    
    # 匹配技能
    skills = await system.match_skills(message)
    
    # 执行技能
    for skill in skills:
        result = await system.execute_skill(skill, {"query": "AI 新闻"})
        print(result)


# 运行
asyncio.run(main())
```

    Write-Host "❌ Docker 未安装" -ForegroundColor Red
    Write-Host "请先下载安装：https://www.docker.com/products/docker-desktop/" -ForegroundColor Cyan
    exit 1

Write-Host "✅ Docker 已安装" -ForegroundColor Green

# 创建虚拟环境
Write-Host "`n📌 创建虚拟环境..." -ForegroundColor Yellow

# 完成

Write-Host "✅ 部署完成！" -ForegroundColor Green

Write-Host "1. 编辑 .env 文件，填入你的 API Key" -ForegroundColor White

Write-Host "3. 或者用 VS Code 打开项目" -ForegroundColor White

## 📦 本地主程序

创建 `main.py`：

```python
"""
🦞 龙虾军团 - 本地运行版本
"""
import os
import asyncio
from flask import Flask, request, jsonify
from flask_socketio import SocketIO
from dotenv import load_dotenv

load_dotenv()

app = Flask(__name__)
socketio = SocketIO(app, cors_allowed_origins="*")

# 全局状态
nodes = {}  # 手机节点状态
agents = {}  # Agent 状态

# ==================== API 接口 ====================

@app.route('/api/health')
def health():
    return {"status": "ok", "nodes": len(nodes)}

@app.route('/api/nodes')
def get_nodes():
    return jsonify(nodes)

@app.route('/api/sensor', methods=['POST'])
def receive_sensor():
    """接收手机节点数据"""
    data = request.json
    node_id = data.get("node_id")
    nodes[node_id] = data
    print(f"📱 收到节点 {node_id} 数据: {data}")
    return {"success": True}

# ==================== 主程序 ====================

def main():
    """启动本地服务"""
    port = int(os.getenv("PORT", 8000))
    
    print("=" * 50)
    print("🦞 龙虾军团本地节点启动")
    print("=" * 50)
    print(f"📡 服务地址: http://localhost:{port}")
    print(f"📱 API 文档: http://localhost:{port}/api")
    print(f"🔌 MQTT: localhost:1883")
    print("=" * 50)
    
    socketio.run(app, host='0.0.0.0', port=port, debug=True)

if __name__ == "__main__":
    main()
```

## 🔗 连接云端服务器（可选）

如果想连接云端服务器，创建 `cloud_client.py`：

```python
"""
连接云端服务器的客户端
"""
import requests
import asyncio
from dotenv import load_dotenv

load_dotenv()

CLOUD_URL = os.getenv("CLOUD_SERVER_URL", "")
CLOUD_KEY = os.getenv("CLOUD_API_KEY", "")

def sync_to_cloud(data: dict):
    """同步数据到云端"""
    if not CLOUD_URL:
        return
    
    try:
        response = requests.post(
            f"{CLOUD_URL}/api/sync",
            json=data,
            headers={"Authorization": f"Bearer {CLOUD_KEY}"}
        )
        return response.json()
    except Exception as e:
        print(f"❌ 同步失败: {e}")

def query_cloud(prompt: str):
    """让云端处理复杂任务"""
    if not CLOUD_URL:
        return "未配置云端服务器"
    
    try:
        response = requests.post(
            f"{CLOUD_URL}/api/query",
            json={"prompt": prompt},
            headers={"Authorization": f"Bearer {CLOUD_KEY}"}
        )
        return response.json().get("response")
    except Exception as e:
        return f"❌ 云端请求失败: {e}"

if __name__ == "__main__":
    # 测试连接
    result = query_cloud("你好")
    print(result)
```

## ❓ 常见问题

### 1. Python 安装失败

- 下载：https://www.python.org/downloads/
- 安装时**务必勾选 "Add Python to PATH"**

### 2. Docker 启动失败

- 打开 Docker Desktop
- 如果显示 WSL 错误，运行：`wsl --install`

### 3. pip 安装超时

- 换清华源：
```bash
pip install -i https://pypi.tuna.tsinghua.edu.cn/simple package-name
```

### 4. 端口被占用

- 改端口，编辑 `.env`：
```
PORT=8001
```

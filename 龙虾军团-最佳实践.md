```python
# ✅ 推荐：动态加载技能
class SkillRegistry:
    """动态技能注册表"""
    
    def __init__(self):
        self.skills = {}
        self._auto_discover()
    
    def _auto_discover(self):
        """自动发现 skills 目录下的技能"""
        import os
        for file in os.listdir("./skills"):
            if file.endswith(".yaml"):
                self._load_skill(f"./skills/{file}")
    
    def _load_skill(self, path: str):
        """加载技能配置"""
        with open(path) as f:
            config = yaml.safe_load(f)
        
        skill = Skill(
            name=config["name"],
            triggers=config.get("triggers", []),
            entry=config["entry"]
        )
        self.skills[skill.name] = skill
```

### 3. 技能执行器

```python
class SkillExecutor:
    """技能执行器 - 带超时、重试、降级"""
    
    def __init__(self):
        self.registry = SkillRegistry()
    
    async def execute(self, skill_name: str, context: dict) -> Result:
        skill = self.registry.get(skill_name)
        
        try:
            # 超时控制
            result = await asyncio.wait_for(
                skill.execute(context),
                timeout=skill.timeout
            )
            return Result(success=True, data=result)
            
        except asyncio.TimeoutError:
            # 超时降级
            return await self._fallback(skill, context, "timeout")
            
        except Exception as e:
            # 错误重试
            return await self._retry(skill, context, e)
```

## 💾 记忆系统最佳实践

### 1. 三层记忆架构

```python
# ✅ 推荐：三层记忆
class ThreeLayerMemory:
    """三层记忆 - 永不丢失"""
    
    def __init__(self):
        # Layer 1: Redis - 短期高频
        self.redis = redis.Redis(
            host='localhost', 
            port=6379, 
            decode_responses=True
        )
        
        # Layer 2: Qdrant - 长期语义
        self.qdrant = QdrantClient(url="http://localhost:6333")
        self._ensure_collection()
        
        # Layer 3: SQLite - 持久化备份
        self.db = sqlite3.connect("memory.db")
        self._ensure_table()
    
    async def remember(self, key: str, content: str, metadata: dict = None):
        """写记忆 - 自动分层"""
        
        # 1. 短期：Redis (24小时 TTL)
        self.redis.setex(f"short:{key}", 86400, json.dumps({
            "content": content,
            "metadata": metadata
        }))
        
        # 2. 长期：Qdrant 向量存储
        vector = await self._embed(content)
        self.qdrant.upsert(
            collection_name="longterm",
            points=[{
                "id": key,
                "vector": vector,
                "payload": {"content": content, "metadata": metadata}
            }]
        )
        
        # 3. 持久：SQLite 重要数据
        if metadata and metadata.get("important"):
            self.db.execute(
                "INSERT OR REPLACE INTO memory VALUES (?, ?, ?, ?)",
                (key, content, json.dumps(metadata), datetime.now())
            )
            self.db.commit()
    
    async def recall(self, query: str, limit: int = 5) -> list:
        """语义检索记忆"""
        
        vector = await self._embed(query)
        
        results = self.qdrant.search(
            collection_name="longterm",
            query_vector=vector,
            limit=limit
        )
        
        return [r.payload["content"] for r in results]
```

### 1. 意图预判

```python
# ✅ 推荐：多层意图识别
class IntentPredictor:
    """意图预判器"""
    
    def __init__(self, llm, skills: list[Skill]):
        self.llm = llm
        self.skills = {s.name: s for s in skills}
    
    async def predict(self, message: str, context: dict) -> list[Skill]:
        # Layer 1: 关键词快速匹配
        keyword_skills = self._keyword_match(message)
        if keyword_skills:
            return keyword_skills
        
        # Layer 2: 意图分类器
        intent = await self._classify_intent(message)
        if intent:
            return self._get_skills_for_intent(intent)
        
        # Layer 3: LLM 语义理解
        semantic_skills = await self._semantic_match(message, context)
        return semantic_skills
    
    def _keyword_match(self, message: str) -> list[Skill]:
        """快速关键词匹配"""
        skills = []
        for skill in self.skills.values():
            for kw in skill.triggers.get("keywords", []):
                if kw in message:
                    skills.append(skill)
                    break
        return skills
```

```python
# ✅ 推荐：事件触发器
class EventDrivenTrigger:
    """事件驱动触发"""
    
    def __init__(self):
        self.handlers = defaultdict(list)
    
    def register(self, event: str, handler: callable):
        self.handlers[event].append(handler)
    
    async def trigger(self, event: str, data: dict):
        for handler in self.handlers[event]:
            await handler(data)


# 使用示例
```

### 3. 监控系统

```python
# ✅ 推荐：主动监控
class ProactiveMonitor:
    """主动监控系统"""
    
    def __init__(self):
        self.monitors = []
    
    def add(self, name: str, interval: int, check_fn, skills: list[Skill]):
        self.monitors.append({
            "name": name,
            "interval": interval,
            "check": check_fn,
            "skills": skills
        })
    
    async def start(self):
        tasks = []
        for m in self.monitors:
            tasks.append(asyncio.create_task(self._run(m)))
        await asyncio.gather(*tasks)
    
    async def _run(self, monitor):
        while True:
            result = await monitor["check"]()
            if result.get("alert"):
                for skill in monitor["skills"]:
                    await skill.execute(result)
            await asyncio.sleep(monitor["interval"])
```

## ⚠️ 错误处理与容错

### 1. 优雅降级

```python
# ✅ 推荐：多级降级
class GracefulDegradation:
    """优雅降级"""
    
    def __init__(self):
        self.fallbacks = {}
    
    def register(self, capability: str, fallbacks: list[callable]):
        """注册降级函数"""
        self.fallbacks[capability] = fallbacks
    
    async def execute(self, capability: str, *args, **kwargs):
        """执行带降级"""
        for fallback in self.fallbacks.get(capability, []):
            try:
                return await fallback(*args, **kwargs)
            except:
                continue
        
        # 最后降级：返回缓存或默认响应
        return self._get_default(capability)
```

### 2. 重试策略

```python
# ✅ 推荐：指数退避重试
async def retry_with_backoff(fn, max_retries=3, base_delay=1):
    """指数退避重试"""
    for attempt in range(max_retries):
        try:
            return await fn()
        except Exception as e:
            if attempt == max_retries - 1:
                raise
            
            delay = base_delay * (2 ** attempt)
            await asyncio.sleep(delay)
    
    raise Exception("Max retries exceeded")
```

```python
class MultiLevelCache:
    """多级缓存"""
    
    def __init__(self):
        self.l1 = {}  # 内存
        self.l2 = redis.Redis()  # Redis
    
    async def get(self, key: str):
        # L1
        if key in self.l1:
            return self.l1[key]
        
        # L2
        value = self.l2.get(key)
        if value:
            self.l1[key] = value
            return value
        
        return None
```

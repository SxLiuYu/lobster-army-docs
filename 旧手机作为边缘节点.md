#### 创建感知脚本

```python
# ~/sensor_node.py
#!/usr/bin/env python3
import json
import time
import subprocess
import paho.mqtt.client as mqtt
from datetime import datetime

# 配置
NODE_ID = "phone_living_room"  # 节点名称
BROKER = "192.168.1.x"  # 你家服务器的 IP
MQTT_USER = "lobster"
MQTT_PASS = "password"

# MQTT 连接
client = mqtt.Client(client_id=NODE_ID)
client.username_pw_set(MQTT_USER, MQTT_PASS)
client.connect(BROKER, 1883, 60)
client.loop_start()

def get_sensor_data():
    """获取传感器数据"""
    data = {}
    
    # 电池状态
    try:
        result = subprocess.run(
            ["termux-battery-status"],
            capture_output=True, text=True
        )
        if result.returncode == 0:
            battery = json.loads(result.stdout)
            data["battery"] = battery.get("percentage")
            data["charging"] = battery.get("plugged") == "PLUGGED"
    except:
        pass
    
    # 光线传感器
    try:
        result = subprocess.run(
            ["termux-sensor"],
            capture_output=True, text=True
        )
        # 解析传感器数据
    except:
        pass
    
    # 网络状态
    try:
        result = subprocess.run(
            ["termux-wifi-connectioninfo"],
            capture_output=True, text=True
        )
        if result.returncode == 0:
            wifi = json.loads(result.stdout)
            data["wifi_ssid"] = wifi.get("ssid", "Disconnected")
    except:
        pass
    
    return data

def main():
    """主循环"""
    print(f"🚀 节点 {NODE_ID} 启动")
    
    while True:
        try:
            # 获取数据
            sensor_data = get_sensor_data()
            
            # 添加时间戳
            data = {
                "node_id": NODE_ID,
                "timestamp": datetime.now().isoformat(),
                "sensors": sensor_data
            }
            
            # 发送 MQTT
            client.publish(
                f"lobster/{NODE_ID}/status",
                json.dumps(data)
            )
            
            print(f"📤 已发送: {data}")
            
        except Exception as e:
            print(f"❌ 错误: {e}")
        
        # 每 30 秒发送一次
        time.sleep(30)

if __name__ == "__main__":
    main()
```

### 2. iOS 端（使用 Pythonista）

如果是有越狱的 iPhone，可以用 Pythonista：

```python
# Pythonista 脚本
import urllib.request
import json
import location

# 配置
SERVER_URL = "http://192.168.1.x:8765/api/sensor"

def send_data():
    # 获取位置
    location.start_updates()
    loc = location.get_last_location()
    
    data = {
        "node_id": "iphone_bedroom",
        "location": {"lat": loc.latitude, "lng": loc.longitude},
        "timestamp": time.time()
    }
    
    # 发送
    req = urllib.request.Request(
        SERVER_URL,
        data=json.dumps(data).encode(),
        headers={"Content-Type": "application/json"}
    )
    urllib.request.urlopen(req)

# 定时执行
import time
while True:
    send_data()
    time.sleep(60)
```

### 3. 服务器端接收

```python
# server.py - 接收节点数据
from flask import Flask, request, jsonify
from flask_socketio import SocketIO

app = Flask(__name__)
socketio = SocketIO(app)

# 存储节点状态
nodes = {}

@app.route('/api/sensor', methods=['POST'])
def receive_sensor():
    data = request.json
    
    node_id = data.get("node_id")
    nodes[node_id] = data
    
    # 广播给 Agent
    socketio.emit('node_update', data)
    
    return {"success": True}

@app.route('/api/nodes')
def get_nodes():
    return jsonify(nodes)

@socketio.on('connect')
def handle_connect():
    print("Agent 已连接")

if __name__ == '__main__':
    socketio.run(app, host='0.0.0.0', port=8765)
```

## 🎯 实际应用场景


```
手机检测到断开WiFi + 移动到某位置 →

```

### 场景 3：异常检测

```
客厅摄像头检测到异常移动 →
```

## 🎯 核心问题

### 当前模式（被动）

```
用户 → 发起请求 → Agent → 响应
```

**问题**：用户必须明确知道需要什么技能，每次都要主动要求。

### 目标模式（主动）

```
用户 → 发起请求 → Agent → 自主分析 → 自动调用技能 → 响应
                                    ↓
                              [主动发现需求]
                                    ↓
                              [自动执行任务]
```

## 🚀 自主触发机制

### 1. 意图预判（Intent Prediction）

```python
class IntentPredictor:
    """意图预测器 - 在执行前预判用户可能需要的技能"""
    
    def __init__(self, skills: list[Skill], llm):
        self.skills = skills
        self.llm = llm
    
    def predict(self, user_message: str) -> list[Skill]:
        """预测用户可能需要的技能"""
        
        # 构建 prompt
        prompt = f"""
        用户消息: "{user_message}"
        
        可用技能:
        {self._format_skills()}
        
        分析用户可能的意图，列出需要自动调用的技能。
        返回技能名称列表，用逗号分隔。
        """
        
        response = self.llm.predict(prompt)
        return self._parse_skills(response)
    
    def _format_skills(self) -> str:
        return "\n".join([
            f"- {s.name}: {s.description}" 
            for s in self.skills
        ])
```

### 2. 上下文感知（Context Awareness）

```python
class ContextAwareSkillManager:
    """上下文感知的技能管理器"""
    
    def __init__(self, skill_registry, memory):
        self.registry = skill_registry
        self.memory = memory
    
    def get_relevant_skills(self, task: str, context: dict) -> list[Skill]:
        """根据任务和上下文获取相关技能"""
        
        relevant = []
        
        # 1. 关键词匹配
        keyword_skills = self._match_by_keywords(task)
        relevant.extend(keyword_skills)
        
        # 2. 历史经验
        history_skills = self._match_by_history(task, context)
        relevant.extend(history_skills)
        
        # 3. 时间触发
        time_skills = self._match_by_time(context)
        relevant.extend(time_skills)
        
        # 4. 环境触发
        env_skills = self._match_by_environment(context)
        relevant.extend(env_skills)
        
        return self._deduplicate(relevant)
    
    def _match_by_keywords(self, task: str) -> list[Skill]:
        """关键词匹配"""
        skills = []
        task_lower = task.lower()
        
        keywords = {
            "search": [WebSearchTool()],
            "写代码": [CodeExecutorTool()],
            "文档": [DocumentWriterTool()],
            "部署": [DeployTool()],
            "监控": [MonitorTool()],
            "搜索": [WebSearchTool()],
            "分析": [AnalysisTool()],
        }
        
        for keyword, skill_list in keywords.items():
            if keyword in task_lower:
                skills.extend(skill_list)
        
        return skills
```

### 3. 事件触发（Event-Driven）

```python
class EventDrivenSkillTrigger:
    """事件驱动的技能触发器"""
    
    def __init__(self):
        self.listeners = {}
    
    def register_event(self, event_name: str, skill: Skill, condition: callable):
        """注册事件触发器"""
        if event_name not in self.listeners:
            self.listeners[event_name] = []
        
        self.listeners[event_name].append({
            "skill": skill,
            "condition": condition
        })
    
    def trigger(self, event_name: str, event_data: dict) -> list[Skill]:
        """触发事件并返回需要执行的技能"""
        
        triggered = []
        
        for listener in self.listeners.get(event_name, []):
            if listener["condition"](event_data):
                triggered.append(listener["skill"])
        
        return triggered


# 使用示例
trigger = EventDrivenSkillTrigger()

# 注册事件：每天早上自动执行

    "daily_morning",

)

# 注册事件：发现异常时执行
trigger.register_event(
    "anomaly_detected",
    skill=AlertSkill(),
    condition=lambda ctx: ctx.get("severity") == "high"
)
```

## 🛠️ 技能自动发现与加载

### 动态技能注册

```python
class DynamicSkillLoader:
    """动态技能加载器"""
    
    def __init__(self, skills_dir: str = "./skills"):
        self.skills_dir = skills_dir
        self.registry = {}
        self._auto_discover()
    
    def _auto_discover(self):
        """自动发现并加载技能"""
        
        import os
        import importlib.util
        import yaml
        
        for filename in os.listdir(self.skills_dir):
            if filename.endswith((".py", ".yaml")):
                path = os.path.join(self.skills_dir, filename)
                self._load_skill(path)
    
    def _load_skill(self, path: str):
        """加载单个技能"""
        
        if path.endswith(".yaml"):
            # YAML 格式配置
            with open(path) as f:
                config = yaml.safe_load(f)
                
            skill = Skill(
                name=config["name"],
                description=config["description"],
                version=config.get("version", "1.0.0"),
                triggers=config.get("triggers", []),
                capabilities=config.get("capabilities", []),
                entry_point=config.get("entry_point")
            )
            
        elif path.endswith(".py"):
            # Python 代码
            spec = importlib.util.spec_from_file_location("skill", path)
            module = importlib.util.module_from_spec(spec)
            spec.loader.exec_module(module)
            
            if hasattr(module, "Skill"):
                skill = module.Skill()
            else:
                return
        
        self.registry[skill.name] = skill
        print(f"✅ 加载技能: {skill.name}")
    
    def get_skill(self, name: str) -> Skill:
        """获取技能"""
        return self.registry.get(name)
    
    def list_skills(self) -> list[Skill]:
        """列出所有技能"""
        return list(self.registry.values())
    
    def find_skills(self, query: str) -> list[Skill]:
        """查找相关技能"""
        query_lower = query.lower()
        return [
            s for s in self.registry.values()
            if query_lower in s.name.lower() or 
               query_lower in s.description.lower()
        ]
```

## 🎯 意图识别与路由

### 多层意图识别

```python
class IntentRouter:
    """意图路由器 - 智能判断需要什么技能"""
    
    def __init__(self, skill_registry, memory):
        self.registry = skill_registry
        self.memory = memory
    
    def route(self, message: str, context: dict = None) -> RouteDecision:
        """路由决策"""
        
        context = context or {}
        
        # Layer 1: 精确匹配
        exact = self._exact_match(message)
        if exact:
            return RouteDecision(
                intent="exact",
                skills=exact,
                confidence=1.0
            )
        
        # Layer 2: 关键词匹配
        keywords = self._keyword_match(message)
        if keywords:
            return RouteDecision(
                intent="keyword",
                skills=keywords,
                confidence=0.8
            )
        
        # Layer 3: 语义匹配（LLM）
        semantic = await self._semantic_match(message, context)
        if semantic:
            return RouteDecision(
                intent="semantic",
                skills=semantic["skills"],
                confidence=semantic["confidence"],
                reasoning=semantic["reasoning"]
            )
        
        # Layer 4: 上下文推断
        context_infer = self._context_infer(message, context)
        if context_infer:
            return RouteDecision(
                intent="context",
                skills=context_infer,
                confidence=0.6
            )
        
        # Default: 只返回文本响应
        return RouteDecision(
            intent="default",
            skills=[],
            confidence=0.0
        )
    
    async def _semantic_match(self, message: str, context: dict) -> dict:
        """LLM 语义匹配"""
        
        # 获取可用技能
        skills = self.registry.list_skills()
        skills_desc = "\n".join([
            f"- {s.name}: {s.description}" 
            for s in skills
        ])
        
        # 构建 prompt
        prompt = f"""
        用户消息: "{message}"
        
        可用技能:
        {skills_desc}
        
        上下文: {context}
        
        分析用户意图，判断需要调用哪些技能。
        返回 JSON 格式:
        {{
            "skills": ["skill_name1", "skill_name2"],
            "confidence": 0.95,
            "reasoning": "简短分析"
        }}
        
        如果不需要任何技能，返回空的 skills 数组。
        """
        
        response = await self.llm.agenerate(prompt)
        result = json.loads(response)
        
        # 转换为 Skill 对象
        skill_objects = [
            self.registry.get_skill(name) 
            for name in result["skills"]
            if self.registry.get_skill(name)
        ]
        
        return {
            "skills": skill_objects,
            "confidence": result["confidence"],
            "reasoning": result["reasoning"]
        }
```

### 路由决策结果

```python
@dataclass
class RouteDecision:
    """路由决策"""
    intent: str                    # 匹配类型
    skills: list[Skill]           # 需要调用的技能
    confidence: float             # 置信度
    reasoning: str = ""           # 分析理由
    should_respond: bool = True   # 是否需要文本回复
    
    def __post_init__(self):
        # 如果有技能需要调用，可能不需要立即回复
        self.should_respond = len(self.skills) == 0
```

## ⏰ 案例：自主监控与预警

### 场景：Agent 自主监控并处理问题

```python
class ProactiveMonitor:
    """主动监控 - 不需要用户要求就能发现问题并处理"""
    
    def __init__(self, skill_registry):
        self.registry = skill_registry
        self.monitors = []
        self.running = False
    
    def add_monitor(self, name: str, interval: int, check_fn: callable, skills: list[Skill]):
        """添加监控任务"""
        self.monitors.append({
            "name": name,
            "interval": interval,
            "check": check_fn,
            "skills": skills,
            "last_check": None
        })
    
    async def start(self):
        """启动所有监控"""
        self.running = True
        
        tasks = []
        for monitor in self.monitors:
            task = asyncio.create_task(
                self._run_monitor(monitor)
            )
            tasks.append(task)
        
        await asyncio.gather(*tasks)
    
    async def _run_monitor(self, monitor: dict):
        """运行单个监控"""
        
        while self.running:
            try:
                # 执行检查
                result = await monitor["check"]()
                
                if result.get("trigger"):
                    # 触发技能
                    for skill in monitor["skills"]:
                        await skill.execute(result["data"])
                        
            except Exception as e:
                print(f"Monitor error: {e}")
            
            await asyncio.sleep(monitor["interval"])


# 使用示例
monitor = ProactiveMonitor(skill_registry)

# 添加系统监控
monitor.add_monitor(
    name="system_health",
    interval=60,  # 每 60 秒
    check_fn=lambda: check_system_health(),
    skills=[AlertSkill(), AutoFixSkill()]
)

# 添加邮件监控
monitor.add_monitor(
    name="email_important",
    interval=300,  # 每 5 分钟
    check_fn=lambda: check_important_emails(),
    skills=[EmailSummarySkill(), PrioritySortSkill()]
)

# 添加新闻监控
monitor.add_monitor(
    name="news_trending",
    interval=3600,  # 每小时
    check_fn=lambda: check_trending_news(),
    skills=[DigestSkill(), ShareSkill()]
)
```

### 自动执行工作流

```python
class AutoExecutionEngine:
    """自动执行引擎"""
    
    def __init__(self, skill_registry, memory):
        self.registry = skill_registry
        self.memory = memory
    
    async def execute(self, task: str, context: dict = None):
        """执行任务 - 自动选择和调用技能"""
        
        context = context or {}
        
        # 1. 规划执行步骤
        plan = await self._plan_execution(task, context)
        
        # 2. 逐步执行
        results = []
        for step in plan:
            skill = self.registry.get_skill(step["skill"])
            
            # 执行技能
            result = await skill.execute(
                step["input"],
                context={**context, "previous_results": results}
            )
            
            results.append(result)
            
            # 检查是否需要调整后续步骤
            if self._should_adjust(plan, results):
                plan = await self._replan(task, context, results)
        
        # 3. 汇总结果
        return self._aggregate_results(results)
    
    async def _plan_execution(self, task: str, context: dict) -> list[dict]:
        """规划执行步骤"""
        
        skills = self.registry.list_skills()
        
        prompt = f"""
        任务: "{task}"
        上下文: {context}
        
        可用技能:
        {json.dumps([s.to_dict() for s in skills])}
        
        规划执行步骤，返回 JSON 数组:
        [
            {{"step": 1, "skill": "skill_name", "input": "描述输入"}},
            {{"step": 2, "skill": "skill_name", "input": "描述输入"}}
        ]
        """
        
        response = await self.llm.agenerate(prompt)
        return json.loads(response)
```

```python
# main.py
import asyncio
from dataclasses import dataclass
from typing import Optional

class LobsterArmyProactive:
    """🦞 主动式龙虾军团"""
    
    def __init__(self):
        # 1. 初始化技能注册表
        self.skill_registry = DynamicSkillLoader("./skills")
        
        # 2. 初始化记忆系统
        self.memory = AgentMemory()
        
        # 3. 初始化意图路由
        self.router = IntentRouter(self.skill_registry, self.memory)
        
        # 4. 初始化监控
        self.monitor = ProactiveMonitor(self.skill_registry)
        
        # 5. 配置 LLM
        self.llm = ChatOpenAI(model="gpt-4")
    
    async def process(self, message: str, context: dict = None):
        """处理用户消息 - 自主判断需要的技能"""
        
        context = context or {}
        
        # Step 1: 路由决策
        decision = await self.router.route(message, context)
        
        print(f"🎯 意图: {decision.intent} (置信度: {decision.confidence})")
        print(f"🛠️ 需要技能: {[s.name for s in decision.skills]}")
        
        # Step 2: 执行技能
        if decision.skills:
            results = []
            for skill in decision.skills:
                result = await skill.execute(
                    message, 
                    context={**context, "memory": self.memory}
                )
                results.append(result)
            
            # Step 3: 汇总结果
            return self._format_response(results)
        else:
            # Step 4: 直接回复
            return await self._generate_direct_response(message)
    
    async def start_monitoring(self):
        """启动主动监控"""
        
        # 添加各种监控任务
        self.monitor.add_monitor(
            "email_check",
            interval=300,
            check_fn=self._check_emails,
            skills=[EmailSkill(), SummarySkill()]
        )
        
        self.monitor.add_monitor(
            "system_check",
            interval=60,
            check_fn=self._check_system,
            skills=[AlertSkill()]
        )
        
        # 启动
        await self.monitor.start()
    
    def _format_response(self, results: list) -> str:
        """格式化结果"""
        
        parts = []
        for r in results:
            if r.get("type") == "text":
                parts.append(r["content"])
            elif r.get("type") == "data":
                parts.append(f"📊 数据: {r['summary']}")
            elif r.get("type") == "alert":
                parts.append(f"⚠️ 提醒: {r['message']}")
        
        return "\n\n".join(parts)


# 使用
army = LobsterArmyProactive()

# 处理用户消息
result = await army.process("帮我搜一下最新的 AI 新闻")
print(result)

# 启动主动监控
await army.start_monitoring()
```

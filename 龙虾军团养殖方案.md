## 💾 记忆不丢失方案

### 三层记忆架构

```python
from dataclasses import dataclass
from typing import Any, Optional
import json
import redis
from qdrant_client import QdrantClient

@dataclass
class AgentMemory:
    """三层记忆系统 - 永不丢失"""
    
    # 短期：Redis（最近会话，TTL 24小时）
    short_term: redis.Redis
    
    # 长期：Qdrant（向量存储）
    long_term: QdrantClient
    
    # 持久：文件系统（重要记忆落盘）
    persistent_path: str
    
    def remember(self, key: str, value: Any, important: bool = False):
        """写入记忆 - 自动分层"""
        # 1. 写入短期记忆
        self.short_term.setex(f"mem:{key}", 86400, json.dumps(value))
        
        # 2. 写入向量库（支持语义检索）
        self.long_term.upsert(
            collection_name="agent_memory",
            points=[{
                "id": key,
                "vector": self._embed(value),
                "payload": {"content": value}
            }]
        )
        
        # 3. 重要记忆落盘
        if important:
            with open(f"{self.persistent_path}/{key}.json", "w") as f:
                json.dump({"key": key, "value": value}, f)
    
    def recall(self, query: str) -> list:
        """语义检索记忆"""
        results = self.long_term.search(
            collection_name="agent_memory",
            query_vector=self._embed(query),
            limit=5
        )
        return [r.payload["content"] for r in results]
```

### Skill Registry

```python
class SkillRegistry:
    """技能市场 - 动态注册与加载"""
    
    def __init__(self):
        self.skills: dict[str, Skill] = {}
        
    def register(self, skill: Skill):
        """注册技能"""
        self.skills[skill.name] = skill
        print(f"✅ 注册技能: {skill.name} v{skill.version}")
        
    def get_skill(self, name: str) -> Skill:
        """获取技能"""
        return self.skills.get(name)
        
    def list_skills(self) -> list[Skill]:
        """列出所有技能"""
        return list(self.skills.values())
```

## 🤖 多 Agent 协作模式

### 1. 指挥官模式

```python
from autogen import ConversableAgent, GroupChat, GroupChatManager

commander = ConversableAgent(
    name="Commander",
    system_message="你是指挥官，负责分发任务给各 Agent",
    llm_config=llm_config
)

researcher = ConversableAgent(
    name="Researcher", 
    system_message="你负责信息收集",
    llm_config=llm_config
)

coder = ConversableAgent(
    name="Coder",
    system_message="你负责写代码",
    llm_config=llm_config
)

# 组队
group_chat = GroupChat(
    agents=[commander, researcher, coder],
    messages=[],
    max_round=5
)

manager = GroupChatManager(groupchat=group_chat)
```

### 2. CrewAI 模式

```python
from crewai import Agent, Task, Crew

# 定义 Agent
researcher = Agent(
    role="研究员",
    goal="收集最准确的信息",
    backstory="你是一个专业的研究员"
)

coder = Agent(
    role="工程师", 
    goal="写出高质量代码",
    backstory="你是一个资深工程师"
)

# 定义任务
task1 = Task(description="研究内网穿透方案", agent=researcher)
task2 = Task(description="实现代码", agent=coder)

# crew 执行
crew = Crew(agents=[researcher, coder], tasks=[task1, task2])
result = crew.kickoff()
```

## 📊 Agent 状态监控

```python
@dataclass
class AgentState:
    agent_id: str
    status: str  # idle, working, error
    skills: list[str]
    memory_size: int
    uptime: float
    
    def to_dict(self):
        return {
            "agent_id": self.agent_id,
            "status": self.status,
            "skills": self.skills,
            "memory_entries": self.memory_size,
            "uptime_seconds": self.uptime
        }
```

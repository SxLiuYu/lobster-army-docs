### 1. 天猫精灵语音技能

创建自定义技能来处理语音命令：

```python
# skills/tmall_genie_skill.py
from crewai import Tool

class TmallGenieSkill:
    """天猫精灵技能"""
    
    def __init__(self, home_assistant_url: str):
        self.ha_url = home_assistant_url
        self.name = "tmall_genie_control"
        self.description = "控制智能家居设备"
        self.triggers = {
            "keywords": [
                "开灯", "关灯", "调亮", "调暗",
                "打开空调", "关闭空调", "调高温度",
                "拉开窗帘", "关闭窗帘",
                "打开插座", "关闭插座"
            ],
            "intents": ["smart_home_control"]
        }
    
    async def execute(self, command: str, device: str = None) -> dict:
        """执行控制命令"""
        
        # 解析命令
        action = self._parse_action(command)
        
        # 调用 HomeAssistant API
        result = await self._call_ha(action, device)
        
        return result
    
    def _parse_action(self, command: str) -> dict:
        """解析语音命令"""
        
        commands = {
            "开灯": {"entity": "light", "service": "turn_on"},
            "关灯": {"entity": "light", "service": "turn_off"},
            "调亮": {"entity": "light", "service": "turn_on", "brightness_pct": 100},
            "调暗": {"entity": "light", "service": "turn_on", "brightness_pct": 30},
            "打开空调": {"entity": "climate", "service": "turn_on"},
            "关闭空调": {"entity": "climate", "service": "turn_off"},
            "拉开窗帘": {"entity": "cover", "service": "open_cover"},
            "关闭窗帘": {"entity": "cover", "service": "close_cover"},
        }
        
        for key, value in commands.items():
            if key in command:
                return value
        
        return {"service": "unknown"}
    
    async def _call_ha(self, action: dict, device: str = None) -> dict:
        """调用 HomeAssistant API"""
        
        import aiohttp
        
        url = f"{self.ha_url}/api/services/{action.get('entity', 'homeassistant')}/{action.get('service')}"
        
        payload = {}
        if device:
            payload["entity_id"] = device
        
        async with aiohttp.ClientSession() as session:
            async with session.post(
                url, 
                json=payload,
                headers={"Authorization": "Bearer YOUR_TOKEN"}
            ) as response:
                return await response.json()


# 转换为 CrewAI Tool
def get_tmall_genie_tool(ha_url: str):
    return Tool(
        name="tmall_genie_control",
        description="控制智能家居设备：灯光、空调、窗帘、插座等",
        func=lambda params: TmallGenieSkill(ha_url).execute(**params)
    )
```

### 2. 意图识别与路由

```python
# skills/smart_home_router.py
class SmartHomeRouter:
    """智能家居意图路由器"""
    
    def __init__(self, ha_url: str):
        self.ha_url = ha_url
    
    async def route(self, message: str) -> dict:
        """路由智能家居命令"""
        
        # 关键词匹配
        if any(kw in message for kw in ["灯", "空调", "窗帘", "插座"]):
            return await self._handle_control(message)
        
        # 状态查询
        if any(kw in message for kw in ["温度", "湿度", "状态", "开了吗"]):
            return await self._handle_query(message)
        
        # 场景模式
        if any(kw in message for kw in ["模式", "场景", "回家", "出门"]):
            return await self._handle_scene(message)
        
        return None
    
    async def _handle_control(self, message: str) -> dict:
        """处理控制命令"""
        
        # 使用 LLM 解析
        result = await self.llm.predict(f"""
            分析用户语音命令，提取设备控制信息。
            
            用户命令: {message}
            
            返回 JSON:
            {{
                "device_type": "light/switch/climate/cover",
                "device_name": "设备名称或位置",
                "action": "turn_on/turn_off/set_brightness/set_temperature",
                "value": 目标值(可选)
            }}
        """)
        
        return json.loads(result)
    
    async def _handle_query(self, message: str) -> dict:
        """处理查询命令"""
        
        # 获取设备状态
        url = f"{self.ha_url}/api/states"
        states = await self._get_ha_states()
        
        # 提取相关信息
        relevant = []
        for state in states:
            if "temperature" in state.get("attributes", {}):
                relevant.append(state)
            if "humidity" in state.get("attributes", {}):
                relevant.append(state)
        
        return {"devices": relevant}
    
    async def _handle_scene(self, message: str) -> dict:
        """处理场景命令"""
        
        scenes = {
            "回家": ["开灯", "打开空调", "拉开窗帘"],
            "出门": ["关灯", "关闭空调", "关闭窗帘"],
            "睡觉": ["关灯", "拉上窗帘", "关闭空调"],
            "电影": ["调暗灯光", "关闭窗帘"]
        }
        
        for scene, actions in scenes.items():
            if scene in message:
                return {"scene": scene, "actions": actions}
        
        return None
```

## 🚀 部署配置

### 1. Docker Compose 完整配置

```yaml
version: '3.8'

services:
  # HomeAssistant
  homeassistant:
    image: homeassistant/home-assistant:stable
    container_name: lobster-homeassistant
    ports:
      - "8123:8123"
    volumes:
      - ./ha-config:/config
      - /run/dbus:/run/dbus
    network_mode: host
    privileged: true
    restart: unless-stopped

  # MQTT Broker
  mqtt:
    image: eclipse-mosquitto:2
    container_name: lobster-mqtt
    ports:
      - "1883:1883"
      - "9001:9001"
    volumes:
      - ./mosquitto.conf:/mosquitto/config/mosquitto.conf
    restart: unless-stopped

  # 龙虾军团 Agent
  lobster-agent:
    build: ./src
    container_name: lobster-agent
    ports:
      - "8000:8000"
    environment:
      - HA_URL=http://homeassistant:8123
      - MQTT_URL=mqtt://mqtt:1883
      - OPENAI_API_KEY=${OPENAI_API_KEY}
    depends_on:
      - homeassistant
      - mqtt
    restart: unless-stopped
```

### 2. HomeAssistant 配置

```yaml
# configuration.yaml

# 启用必要的集成
homeassistant:
  name: Home

  longitude: 120.0  # 你的经度
  elevation: 0
  unit_system: metric
  time_zone: Asia/Shanghai

# 灯光

      living_room_light:
        friendly_name: 客厅灯
      bedroom_light:
        friendly_name: 卧室灯

# 空调（示例）
climate:
  - platform: generic_thermostat
    name: 客厅空调
    heater: switch.ac_unit
    target_sensor: sensor.temperature

# 传感器
sensor:
  - platform: mqtt
    name: 温度传感器
    state_topic: home/sensor/temperature
    unit_of_measurement: "°C"
  - platform: mqtt
    name: 湿度传感器
    state_topic: home/sensor/humidity
    unit_of_measurement: "%"

# 自动化
automation:
  - alias: "人来灯亮"

      - platform: state

        to: "on"

      - service: light.turn_on
```

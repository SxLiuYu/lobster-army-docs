## 💾 记忆不丢失方案

### 三层记忆架构

```python
from dataclasses import dataclass
from typing import Any
import json
import redis
from qdrant_client import QdrantClient
import sqlite3

@dataclass
class AgentMemory:
    """三层记忆系统 - 永不丢失"""
    
    short_term: redis.Redis      # 短期：Redis
    long_term: QdrantClient      # 长期：Qdrant 向量库
    db_path: str                 # 持久：SQLite
    
    def remember(self, key: str, value: Any, important: bool = False):
        """写入记忆 - 自动分层"""
        # 1. 写入短期记忆 (TTL 24小时)
        self.short_term.setex(f"mem:{key}", 86400, json.dumps(value))
        
        # 2. 写入向量库（支持语义检索）
        self.long_term.upsert(
            collection_name="agent_memory",
            points=[{
                "id": key,
                "vector": self._embed(value),
                "payload": {"content": value, "important": important}
            }]
        )
        
        # 3. 重要记忆落盘
        if important:
            conn = sqlite3.connect(self.db_path)
            conn.execute(
                "INSERT OR REPLACE INTO memory (key, value) VALUES (?, ?)",
                (key, json.dumps(value))
            )
            conn.commit()
            conn.close()
    
    def recall(self, query: str, limit: int = 5) -> list:
        """语义检索记忆"""
        results = self.long_term.search(
            collection_name="agent_memory",
            query_vector=self._embed(query),
            limit=limit
        )
        return [r.payload["content"] for r in results]
```

```python
from dataclasses import dataclass
from typing import Callable, Any

@dataclass
class Skill:
    name: str
    description: str
    version: str
    entry_point: Callable  # 可执行函数
    capabilities: list[str]
    tools_needed: list[str]  # 需要的工具
    
    def execute(self, *args, **kwargs) -> Any:
        """执行技能"""
        return self.entry_point(*args, **kwargs)
```

## 🤖 多 Agent 协作模式

### 1. AutoGen 模式（推荐）

```python
from autogen import ConversableAgent, GroupChat, GroupChatManager

# 定义指挥官
commander = ConversableAgent(
    name="Commander",
    system_message="""你是指挥官，负责:
1. 理解用户需求
2. 将任务分发给合适的 Agent
3. 汇总结果返回给用户
4. 管理所有 Agent 的记忆""",
    llm_config={
        "model": "gpt-4",
        "temperature": 0.7,
        "api_key": os.getenv("OPENAI_API_KEY")
    }
)

# 定义各类型 Agent
researcher = ConversableAgent(
    name="Researcher",
    system_message="你负责信息收集和研究。使用搜索工具查找相关信息。",
    llm_config={"model": "gpt-4"}
)

coder = ConversableAgent(
    name="Coder",
    system_message="你负责写代码。根据需求实现功能。",
    llm_config={"model": "gpt-4"}
)

writer = ConversableAgent(
    name="Writer",
    system_message="你负责写作。将信息整理成文档。",
    llm_config={"model": "gpt-4"}
)

ops = ConversableAgent(
    name="Ops",
    system_message="你负责运维。执行命令、管理服务。",
    llm_config={"model": "gpt-4"}
)

# 组队
group_chat = GroupChat(
    agents=[commander, researcher, coder, writer, ops],
    messages=[],
    max_round=10,
    speaker_selection_method="round_robin"
)

manager = GroupChatManager(groupchat=group_chat)

# 使用
result = commander.initiate_chat(
    manager,
    message="帮我研究内网穿透方案，然后写一个实现"
)
```

### 2. CrewAI 模式

```python
from crewai import Agent, Task, Crew, Process

# 定义 Agent
researcher = Agent(
    role="研究员",
    goal="收集最准确的信息",
    backstory="你是一个专业的研究员，擅长搜索和分析",
    tools=[search_tool, browse_tool]
)

coder = Agent(
    role="工程师", 
    goal="写出高质量代码",
    backstory="你是一个资深工程师，注重代码质量和性能"
)

ops = Agent(
    role="运维",
    goal="确保服务稳定运行",
    backstory="你是一个运维专家，擅长自动化和监控"
)

# 定义任务
task1 = Task(
    description="研究 FRP 内网穿透方案",
    agent=researcher,
    expected_output="详细的技术方案文档"
)

task2 = Task(
    description="实现内网穿透客户端",
    agent=coder,
    expected_output="可运行的代码"
)

task3 = Task(
    description="部署和监控服务",
    agent=ops,
    expected_output="部署脚本和监控配置"
)

# 创建 Crew
crew = Crew(
    agents=[researcher, coder, ops],
    tasks=[task1, task2, task3],
    process=Process.hierarchical,  # 层级协作
    manager_agent=commander
)

# 执行
result = crew.kickoff()
```

## 📡 内网穿透部署

### 使用 FRP 实现远程访问

```
┌─────────────┐     公网服务器      ┌─────────────┐
│  家中电脑   │ ◄─────────────────► │  FRP Server │
│ (FRP Client)│   端口映射         │  (VPS)      │
└─────────────┘                    └─────────────┘
        ▲                                    ▲
        │  ┌───────────────────────────────  │
        └──┤  龙虾军团 Agent (可通过FRP访问)  │
           └────────────────────────────────┘
```

### 1. 服务端部署（VPS）

```bash
# 下载 FRP
wget https://github.com/fatedier/frp/releases/download/v0.68.0/frp_0.68.0_linux_amd64.tar.gz
tar -zxf frp_0.68.0_linux_amd64.tar.gz
cd frp_0.68.0_linux_amd64

# 配置 frps.ini
cat > frps.ini << 'EOF'
[common]
bind_port = 7000
dashboard_port = 7500
token = your_secret_token
dashboard_user = admin
dashboard_pwd = your_password
enable_prometheus = true

# TCP 端口范围
allow_ports = 6000-9000
EOF

# 启动
./frps -c frps.ini

# 或使用 Docker
docker run -d --name frps \
  -v ./frps.ini:/frps.ini \
  -p 7000:7000 -p 7500:7500 \
  fatedier/frp:latest \
  frps -c /frps.ini
```

### 2. 客户端部署（家中电脑）

```bash
# 配置 frpc.ini
cat > frpc.ini << 'EOF'
[common]
server_addr = your_vps_ip
server_port = 7000
token = your_secret_token

# SSH
[ssh]
type = tcp
local_ip = 127.0.0.1
local_port = 22
remote_port = 6000

# Web 服务
[web]
type = http
local_ip = 127.0.0.1
local_port = 8080
custom_domains = your-domain.com

# Agent API
[agent-api]
type = tcp
local_ip = 127.0.0.1
local_port = 8000
remote_port = 8000
EOF

# 启动
./frpc -c frpc.ini
```

### 3. 通过 Cloudflare Tunnel 增强安全

```bash
# 安装 cloudflared
wget -q https://github.com/cloudflare/cloudflared/releases/latest/download/cloudflared-linux-amd64 -O /usr/local/bin/cloudflared
chmod +x /usr/local/bin/cloudflared

# 登录
cloudflared tunnel login

cloudflared tunnel create lobster-army

# 配置
cat > tunnel.yaml << 'EOF'
tunnel: lobster-army
credentials-file: /root/.cloudflared/credentials.json

ingress:
  - hostname: agent.yourdomain.com
    service: http://localhost:8000
  - hostname: *.yourdomain.com
    service: http://localhost:8080
  - service: http_status:404
EOF

# 启动
cloudflared tunnel --config tunnel.yaml run
```

## 🚀 快速开始

### 1. 安装依赖

```bash
# Python 环境
python3 -m venv venv
source venv/bin/activate

# 安装核心库
pip install autogen crewai qdrant-client redis fastapi
pip install openai anthropic

# 安装 FRP
wget https://github.com/fatedier/frp/releases/download/v0.68.0/frp_0.68.0_linux_amd64.tar.gz
tar -zxf frp_0.68.0_linux_amd64.tar.gz
```

### 2. 配置环境变量

```bash
# .env
OPENAI_API_KEY=sk-...
ANTHROPIC_API_KEY=sk-ant-...
REDIS_URL=redis://localhost:6379
QDRANT_URL=http://localhost:6333
FRP_SERVER_ADDR=your_vps_ip
FRP_TOKEN=your_token
```

### 3. 初始化项目

```bash
mkdir lobster-army
cd lobster-army

# 创建目录结构
mkdir -p agents skills memory config scripts

# 配置文件
cat > config/agents.yaml << 'EOF'
commander:
  model: gpt-4
  temperature: 0.7
  
researcher:
  model: gpt-4
  tools: [search, browse]
  
coder:
  model: gpt-4
  tools: [code_executor]
  
writer:
  model: gpt-4
  
ops:
  model: gpt-4
  tools: [docker, ssh]
EOF

cat > config/memory.yaml << 'EOF'
short_term:
  type: redis
  ttl: 86400
  
long_term:
  type: qdrant
  collection: agent_memory
  
persistent:
  type: sqlite
  path: memory/memory.db
EOF
```

### 4. 启动龙虾军团

```python
# main.py
import os
from autogen import ConversableAgent
from lobster_army.memory import AgentMemory
from lobster_army.skills import SkillRegistry

# 初始化
memory = AgentMemory(
    short_term=redis.Redis.from_url(os.getenv("REDIS_URL")),
    long_term=QdrantClient(url=os.getenv("QDRANT_URL")),
    db_path="memory/memory.db"
)

skills = SkillRegistry()
skills.auto_discover("skills")

# 启动 Agent
agent = ConversableAgent(
    name="lobster_commander",
    system_message="你是龙虾军团的指挥官...",
    llm_config={"model": "gpt-4", "api_key": os.getenv("OPENAI_API_KEY")}
)

# 开始工作
agent.initiate_chat(
    manager,
    message="帮我收集内网穿透方案，并部署到服务器"
)
```

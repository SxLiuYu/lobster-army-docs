## 🎯 服务质量的本质

### 服务质量金字塔

```
           ┌─────────┐
           │  超预期  │ ← 主动发现需求，提供惊喜服务
           ├─────────┤
           │  个性化  │ ← 记住用户偏好，提供定制服务
           ├─────────┤
           │  连贯性  │ ← 保持上下文，理解对话历史
           ├─────────┤
           │  正确性  │ ← 准确理解意图，正确执行任务
           ├─────────┤
           │  基础响应 │ ← 快速回应，有礼貌
           └─────────┘
```

### 核心指标

| 指标 | 定义 | 目标 |
|------|------|------|
| **响应时间** | 从收到消息到回复 | < 3 秒 |
| **意图准确率** | 正确理解用户意图 | > 90% |
| **任务完成率** | 成功完成用户任务 | > 85% |
| **用户满意度** | 用户反馈正向 | > 4.5/5 |
| **上下文保持** | 多轮对话连贯性 | > 95% |

```python
class ConversationContext:
    """对话上下文管理器"""
    
    def __init__(self, max_turns: int = 10):
        self.max_turns = max_turns
        self.turns = []
        self.entities = {}  # 提取的实体
        self.topics = set()  # 话题
        self.tasks = []  # 任务列表
    
    def add_turn(self, user_message: str, assistant_response: str):
        """添加一轮对话"""
        
        self.turns.append({
            "user": user_message,
            "assistant": assistant_response,
            "timestamp": datetime.now()
        })
        
        # 保持最大轮数
        if len(self.turns) > self.max_turns:
            self.turns.pop(0)
        
        # 提取实体
        self._extract_entities(user_message)
        
        # 更新话题
        self._update_topics(user_message)
    
    def _extract_entities(self, text: str):
        """提取实体（人名、时间、任务等）"""
        
        # 使用 LLM 提取
        prompt = f"""
        从以下文本中提取关键实体：
        
        文本：{text}
        
        返回 JSON：
        {{
            "tasks": ["任务列表"],
            "times": ["时间"],
            "persons": ["人名"],
            "locations": ["地点"],
            "objects": ["物品"]
        }}
        """
        
        result = self.llm.predict(prompt)
        entities = json.loads(result)
        
        for key, values in entities.items():
            if key not in self.entities:
                self.entities[key] = []
            self.entities[key].extend(values)
    
    def get_recent_context(self, turns: int = 3) -> str:
        """获取最近 N 轮上下文"""
        
        recent = self.turns[-turns:]
        context = ""
        
        for turn in recent:
            context += f"用户: {turn['user']}\n"
            context += f"助手: {turn['assistant']}\n"
        
        return context
    
    def get_entity(self, key: str) -> any:
        """获取实体"""
        return self.entities.get(key, [])
    
    def has_ongoing_task(self) -> bool:
        """是否有进行中的任务"""
        return len(self.tasks) > 0 and not self.tasks[-1].get("completed")
```

```python
class TaskTracker:
    """任务追踪器"""
    
    def __init__(self):
        self.tasks = []
    
    def start_task(self, task_description: str, task_id: str = None):
        """开始任务"""
        
        task_id = task_id or generate_id()
        
        task = {
            "id": task_id,
            "description": task_description,
            "status": "in_progress",
            "steps": [],
            "start_time": datetime.now()
        }
        
        self.tasks.append(task)
        return task_id
    
    def update_task(self, task_id: str, step: str, result: any):
        """更新任务进度"""
        
        for task in self.tasks:
            if task["id"] == task_id:
                task["steps"].append({
                    "step": step,
                    "result": result,
                    "timestamp": datetime.now()
                })
                break
    
    def complete_task(self, task_id: str, final_result: any):
        """完成任务"""
        
        for task in self.tasks:
            if task["id"] == task_id:
                task["status"] = "completed"
                task["result"] = final_result
                task["end_time"] = datetime.now()
                break
    
    def get_ongoing_task(self) -> dict:
        """获取进行中的任务"""
        
        for task in reversed(self.tasks):
            if task["status"] == "in_progress":
                return task
        
        return None
    
    def format_progress(self, task_id: str) -> str:
        """格式化任务进度"""
        
        task = self.get_task(task_id)
        if not task:
            return ""
        
        steps = task["steps"]
        progress = f"📋 任务进度：{len(steps)}/??\n\n"
        
        for i, step in enumerate(steps):
            progress += f"{i+1}. {step['step']}\n"
        
        return progress
```

```python
class ErrorHandler:
    """错误处理器"""
    
    ERROR_TYPES = {
        "api_error": {
            "severity": "high",
            "retry": True,
            "max_retries": 3,
            "fallback": "use_cache_or_alternative"
        },
        "timeout": {
            "severity": "medium",
            "retry": True,
            "max_retries": 2,
            "fallback": "notify_user_wait"
        },
        "invalid_input": {
            "severity": "low",
            "retry": False,
            "fallback": "ask_clarification"
        },
        "permission_denied": {
            "severity": "high",
            "retry": False,
            "fallback": "explain_and_suggest"
        },
        "rate_limit": {
            "severity": "medium",
            "retry": True,
            "max_retries": 5,
            "fallback": "queue_and_retry"
        }
    }
    
    def __init__(self, llm):
        self.llm = llm
        self.error_history = []
    
    async def handle(self, error: Exception, context: dict) -> str:
        """处理错误"""
        
        error_type = self._classify_error(error)
        error_info = self.ERROR_TYPES.get(error_type, {})
        
        # 记录错误
        self.error_history.append({
            "type": error_type,
            "error": str(error),
            "timestamp": datetime.now()
        })
        
        # 根据错误类型处理
        if error_info.get("retry") and error_info.get("max_retries", 0) > 0:
            return await self._retry_with_backoff(error, context, error_info)
        elif error_info.get("fallback") == "ask_clarification":
            return await self._ask_clarification(error, context)
        elif error_info.get("fallback") == "use_cache_or_alternative":
            return await self._use_alternative(error, context)
        else:
            return self._default_error_response(error, context)
    
    async def _retry_with_backoff(self, error, context, error_info):
        """带退避的重试"""
        
        max_retries = error_info["max_retries"]
        
        for attempt in range(max_retries):
            try:
                # 指数退避
                wait_time = 2 ** attempt
                await asyncio.sleep(wait_time)
                
                # 重试执行
                return await self._retry_execute(context)
                
            except Exception as e:
                if attempt == max_retries - 1:
                    return await self._fallback_error(e, context)
        
        return self._default_error_response(error, context)
    
    async def _ask_clarification(self, error, context) -> str:
        """请求澄清"""
        
        prompt = f"""
        用户输入导致错误：{str(error)}
        上下文：{context}
        
        生成友好的澄清请求：
        - 说明哪里不清楚
        - 给出具体问题
        - 给出示例
        """
        
        return self.llm.predict(prompt)
    
    def _default_error_response(self, error, context) -> str:
        """默认错误响应"""
        
        error_msg = str(error)[:100]  # 截断错误信息
        
        responses = [
            f"抱歉，我遇到了点问题：{error_msg}。让我换个方式试试。",
            f"出了一点小状况，让我调整一下再试一次。",
            f"这个问题有点复杂，让我换种方法来处理。"
        ]
        
        return random.choice(responses)
```

### 优雅降级

```python
class GracefulDegradation:
    """优雅降级"""
    
    def __init__(self):
        self.fallbacks = {}
    
    def register_fallback(self, capability: str, fallback_fn: callable):
        """注册降级函数"""
        self.fallbacks[capability] = fallback_fn
    
    async def execute_with_fallback(self, capability: str, 
                                     primary_fn, *args, **kwargs):
        """带降级执行"""
        
        try:
            # 尝试主要功能
            return await primary_fn(*args, **kwargs)
            
        except Exception as e:
            # 记录错误
            log.error(f"{capability} failed: {e}")
            
            # 尝试降级
            if capability in self.fallbacks:
                return await self.fallbacks[capability](*args, **kwargs)
            
            # 返回缓存或默认响应
            return await self._get_cached_or_default(capability)
    
    async def _get_cached_or_default(self, capability: str) -> str:
        """获取缓存或默认响应"""
        
        cache = self._get_cache(capability)
        if cache:
            return cache
        
        defaults = {
            "search": "抱歉，搜索服务暂时不可用。我可以帮你做其他事情。",
            "code_execute": "抱歉，代码执行暂时不可用。",
            "web_fetch": "抱歉，无法获取网页内容。"
        }
        
        return defaults.get(capability, "抱歉，服务暂时不可用。")
```

```python
class FeedbackHandler:
    """反馈处理器"""
    
    FEEDBACK_TYPES = {
        "thumbs_up": {
            "weight": 1.0,
            "action": "reinforce",
            "description": "正面反馈"
        },
        "thumbs_down": {
            "weight": -1.0,
            "action": "avoid",
            "description": "负面反馈"
        },
        "correction": {
            "weight": -0.5,
            "action": "learn",
            "description": "纠正"
        },
        "rating": {
            "weight": "rating / 5",
            "action": "adjust",
            "description": "评分"
        }
    }
    
    def __init__(self, user_profile: UserProfile, memory):
        self.profile = user_profile
        self.memory = memory
    
    def process_feedback(self, feedback: dict):
        """处理反馈"""
        
        feedback_type = feedback.get("type")
        feedback_info = self.FEEDBACK_TYPES.get(feedback_type, {})
        
        # 记录反馈
        self.profile.feedback_history.append({
            "type": feedback_type,
            "content": feedback.get("content"),
            "timestamp": datetime.now()
        })
        
        # 根据类型处理
        if feedback_type == "thumbs_up":
            self._reinforce(feedback)
        elif feedback_type == "thumbs_down":
            self._avoid(feedback)
        elif feedback_type == "correction":
            self._learn_correction(feedback)
        elif feedback_type == "rating":
            self._adjust_for_rating(feedback)
    
    def _reinforce(self, feedback: dict):
        """强化正向行为"""
        
        response = feedback.get("response")
        
        # 记录喜欢的回复
        self.profile.liked_responses.append(response)
        
        # 分析为什么这个回复好
        prompt = f"""
        分析以下回复为什么好：
        
        回复：{response}
        
        返回 JSON：
        {{
            "why_good": ["原因1", "原因2"],
            "patterns": ["模式1", "模式2"]
        }}
        """
        
        result = self.llm.predict(prompt)
        
        # 存储分析结果
        self.memory.save_pattern(
            self.profile.user_id,
            "positive",
            json.loads(result)
        )
    
    def _avoid(self, feedback: dict):
        """避免负面行为"""
        
        response = feedback.get("response")
        
        # 记录不喜欢的回复
        self.profile.disliked_responses.append(response)
        
        # 分析为什么不好
        prompt = f"""
        分析以下回复为什么不好：
        
        回复：{response}
        
        返回 JSON：
        {{
            "why_bad": ["原因1", "原因2"],
            "how_to_improve": ["改进1", "改进2"]
        }}
        """
        
        result = self.llm.predict(prompt)
        
        # 存储改进建议
        self.memory.save_pattern(
            self.profile.user_id,
            "negative",
            json.loads(result)
        )
    
    def _learn_correction(self, feedback: dict):
        """从纠正中学习"""
        
        original = feedback.get("original")
        corrected = feedback.get("corrected")
        
        # 存储纠正对
        self.memory.save_correction(
            self.profile.user_id,
            original,
            corrected
        )
```

### 反馈驱动的自适应

```python
class AdaptiveService:
    """自适应服务"""
    
    def __init__(self, feedback_handler: FeedbackHandler, llm):
        self.feedback_handler = feedback_handler
        self.llm = llm
    
    def adapt_response_style(self, base_response: str, 
                           user_feedback: list) -> str:
        """根据反馈调整响应风格"""
        
        if not user_feedback:
            return base_response
        
        # 分析最近的负面反馈
        negatives = [f for f in user_feedback if f.get("type") == "thumbs_down"]
        
        if not negatives:
            return base_response
        
        # 获取改进建议
        suggestions = self._get_improvement_suggestions(negatives)
        
        # 应用调整
        prompt = f"""
        根据以下建议改进回复：
        
        原始回复：{base_response}
        
        改进建议：
        {suggestions}
        
        生成改进后的回复：
        """
        
        return self.llm.predict(prompt)
    
    def _get_improvement_suggestions(self, negatives: list) -> str:
        """获取改进建议"""
        
        suggestions = []
        
        for neg in negatives:
            # 从记忆获取分析结果
            analysis = self.memory.get_negative_pattern(neg.get("response"))
            if analysis:
                suggestions.extend(analysis.get("how_to_improve", []))
        
        return "\n".join([f"- {s}" for s in suggestions])
```

```python
class ServiceMetrics:
    """服务指标"""
    
    def __init__(self):
        self.metrics = {
            "total_requests": 0,
            "successful_requests": 0,
            "failed_requests": 0,
            "average_response_time": 0,
            "intent_accuracy": 0,
            "user_satisfaction": 0
        }
    
    def record_request(self, response_time: float, success: bool, 
                       intent_correct: bool = None):
        """记录请求"""
        
        self.metrics["total_requests"] += 1
        
        if success:
            self.metrics["successful_requests"] += 1
        else:
            self.metrics["failed_requests"] += 1
        
        # 更新平均响应时间
        current_avg = self.metrics["average_response_time"]
        total = self.metrics["total_requests"]
        self.metrics["average_response_time"] = (
            (current_avg * (total - 1) + response_time) / total
        )
    
    def get_success_rate(self) -> float:
        """获取成功率"""
        
        total = self.metrics["total_requests"]
        if total == 0:
            return 0
        
        return self.metrics["successful_requests"] / total
    
    def get_report(self) -> dict:
        """获取报告"""
        
        return {
            "requests": self.metrics["total_requests"],
            "success_rate": self.get_success_rate(),
            "avg_response_time": self.metrics["average_response_time"],
            "user_satisfaction": self.metrics["user_satisfaction"],
            "uptime": self._calculate_uptime()
        }
```

```python
class LobsterArmyService:
    """🦞 龙虾军团服务 - 完整实现"""
    
    def __init__(self, user_id: str):
        # 用户画像
        self.user_profile = await self._load_user_profile(user_id)
        
        # 核心组件
        self.llm = ChatOpenAI(model="gpt-4")
        self.memory = AgentMemory()
        self.context = ConversationContext()
        self.task_tracker = TaskTracker()
        self.error_handler = ErrorHandler(self.llm)
        self.feedback_handler = FeedbackHandler(self.user_profile, self.memory)
        self.metrics = ServiceMetrics()
        
        # 个性化
        self.response_generator = PersonalizedResponseGenerator(
            self.llm, self.user_profile
        )
        
        # 多模态
        self.input_processor = MultimodalInput()
        self.output_generator = MultimodalOutput()
    
    async def process(self, input_data: dict) -> OutputData:
        """处理请求"""
        
        start_time = time.time()
        
        try:
            # 1. 处理输入
            processed = await self.input_processor.process(input_data)
            
            # 2. 检查上下文
            context = self.context.get_recent_context()
            
            # 3. 路由决策
            route = await self.router.route(processed.content, {
                "context": context,
                "user_profile": self.user_profile,
                "ongoing_task": self.task_tracker.get_ongoing_task()
            })
            
            # 4. 执行技能
            if route.skills:
                results = await self._execute_skills(route.skills, processed.content)
                response_text = self._format_response(results)
            else:
                response_text = await self._generate_direct_response(
                    processed.content, context
                )
            
            # 5. 个性化调整
            response_text = self.response_generator.generate(
                response_text, 
                {"input": processed, "route": route}
            )
            
            # 6. 更新上下文
            self.context.add_turn(processed.content, response_text)
            
            # 7. 记录指标
            self.metrics.record_request(
                time.time() - start_time,
                success=True
            )
            
            # 8. 生成输出
            return await self.output_generator.generate(
                Response(content=response_text),
                preferred_mode=input_data.get("mode", "text")
            )
            
        except Exception as e:
            # 错误处理
            error_response = await self.error_handler.handle(e, {
                "input": input_data
            })
            
            self.metrics.record_request(
                time.time() - start_time,
                success=False
            )
            
            return OutputData(
                mode="text",
                content=error_response
            )
    
    async def process_feedback(self, feedback: dict):
        """处理反馈"""
        
        self.feedback_handler.process_feedback(feedback)
        
        # 触发优化
        if len(self.user_profile.feedback_history) >= 10:
            await self._trigger_optimization()
    
    async def get_status(self) -> dict:
        """获取状态"""
        
        return {
            "user": self.user_profile.name,
            "conversation_turns": len(self.context.turns),
            "ongoing_task": self.task_tracker.get_ongoing_task(),
            "metrics": self.metrics.get_report()
        }
```
